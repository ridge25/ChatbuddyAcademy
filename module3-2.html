def load_json_file(file_path):
    if os.path.exists(file_path):
        with open(file_path, "r") as f:
            return json.load(f)
    return {}
        
def save_json_file(data, file_path):
    with open(file_path, "w") as f:
        json.dump(data, f, indent=4)
