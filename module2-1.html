import random
import json
import os
from datetime import datetime
        
# Joke list
jokes = [
    "Why don't skeletons fight each other? They don't have the guts. 💀",
    "What do you call fake spaghetti? An impasta! 🍝",
    "Why did the scarecrow win an award? Because he was outstanding in his field! 🌾",
    "I'm reading a book on anti-gravity. It's impossible to put down! 📚",
    "Why did the math book look sad? Because it had too many problems. ➗"
]
        
# Greeting
print("🤖 Hello! I'm ChatBuddy. What's your name?")
user_name = input("You: ")
        
# Time of day greeting
current_hour = datetime.now().hour
if current_hour < 12:
    time_greeting = "Good morning"
elif current_hour < 18:
    time_greeting = "Good afternoon"
else:
    time_greeting = "Good evening"
        
# Greeting list
greetings = ["Hey there! 👋", "Hiya! 😊", "Howdy! 🤠", "What's up? 🙌", "Yo yo yo! 🤖"]
mood_responses = ["I'm doing great, thanks for asking! 😎", "Better now that you're here! 😁", "Feeling fantastic! What about you? 🌟"]
no_response = [
    "Hmm... I don't know how to respond to that yet. 🤔",
    "I'm still learning! Try something else. 📚",
    "That one's a mystery to me. 😅"
]
        
# Load learned memory
learned_memory_file = "learned_memory.json"
if os.path.exists(learned_memory_file):
    with open(learned_memory_file, "r") as f:
        learned_memory = json.load(f)
else:
    learned_memory = {}
        
# Combined greeting
print(f"ChatBuddy: {time_greeting}, {user_name}! {random.choice(greetings)} Ask me anything. 😄")
print("ChatBuddy: How are you doing today?")
emotion_input = input(f"{user_name}: ").lower()
        
# Joke offer
def offer_joke():
    print("ChatBuddy: Want to hear a joke to cheer you up? (Yes/No)")
    response = input(f"{user_name}: ").lower()
    if "yes" in response:
        print(f"ChatBuddy: {random.choice(jokes)}")
    else:
        print("ChatBuddy: Alright, I'm here if you need a smile later. 🤗")
        
# Emotion response
if "not great" in emotion_input:
    print("ChatBuddy: I'm sorry you're not feeling great. Anything I can do to cheer you up? 🤗")
    offer_joke()
elif "good" in emotion_input or "great" in emotion_input or "happy" in emotion_input:
    print("ChatBuddy: That's awesome to hear! 😄 Keep up the good vibes!")
elif "sad" in emotion_input:
    print("ChatBuddy: I'm sorry you're feeling sad. I'm here if you want to talk. 🤗")
    offer_joke()
elif "tired" in emotion_input:
    print("ChatBuddy: You should try to rest up. Sleep is important! 😴")
elif "hungry" in emotion_input:
    print("ChatBuddy: You better grab a snack! 🍕 Got a favorite food?")
else:
    print("ChatBuddy: Thanks for sharing. I'm here for you no matter what! 😊")
        
# In-memory facts
memory = {}
        
# Main chat loop
while True:
    user_input = input(f"{user_name}: ").lower()
        
    if user_input in learned_memory:
        print(f"ChatBuddy: {learned_memory[user_input]}")
    elif "hello" in user_input or "hi" in user_input:
        print(f"ChatBuddy: {random.choice(greetings)}")
    elif "how are you" in user_input:
        print(f"ChatBuddy: {random.choice(mood_responses)}")
    elif "what time is it" in user_input or "current time" in user_input or "tell me the time" in user_input or "what's the time" in user_input:
        current_time = datetime.now().strftime("%I:%M %p")
        print(f"ChatBuddy: It's currently {current_time} ⏰")
    elif "what's the date" in user_input or "what is the date" in user_input or "current date" in user_input:
        current_date = datetime.now().strftime("%A, %B %d, %Y")
        print(f"ChatBuddy: Today is {current_date} 📅")
    elif "my favorite color is" in user_input:
        color = user_input.split("is")[-1].strip()
        memory["favorite_color"] = color
        print(f"ChatBuddy: Got it! I'll remember your favorite color is {color}. 🎨")
    elif "what's my favorite color" in user_input or "what is my favorite color" in user_input:
        if "favorite_color" in memory:
            print(f"ChatBuddy: Your favorite color is {memory['favorite_color']}! 🌈")
        else:
            print("ChatBuddy: Hmm... I don’t know yet. Tell me your favorite color! 🤷‍♂️")
    elif "my favorite food is" in user_input:
        food = user_input.split("is")[-1].strip()
        memory["favorite_food"] = food
        print(f"ChatBuddy: Yum! I'll remember your favorite food is {food}. 😋")
    elif "what's my favorite food" in user_input or "what is my favorite food" in user_input:
        if "favorite_food" in memory:
            print(f"ChatBuddy: I remember this one! Your favorite food is {memory['favorite_food']}! 🍽️")
        else:
            print("ChatBuddy: Hmm... I don't think you've told me your favorite food yet. Tell me what it is! 🤤")
    elif "my nickname is" in user_input:
        nickname = user_input.split("is")[-1].strip()
        memory["nickname"] = nickname
        print(f"ChatBuddy: That's a cool nickname! I'll remember it—{nickname}, right? 😎")
    elif "what's my nickname" in user_input or "what is my nickname" in user_input:
        if "nickname" in memory:
            print(f"ChatBuddy: You told me this one! Your nickname is {memory['nickname']}! 😄")
        else:
            print("ChatBuddy: Hmm... I don't think you've told me your nickname yet. Wanna share it? 🕵️")
    elif "bye" in user_input:
        print(f"ChatBuddy: Bye {user_name}! Catch you later! 👋")
        break
    else:
        # Emotional keywords
        if any(word in user_input for word in ["good", "great", "sad", "tired", "hungry", "not great"]):
            if "not great" in user_input:
                print("ChatBuddy: Rough day? I'm here to chat. 💬")
                offer_joke()
            elif "good" in user_input or "great" in user_input:
                print("ChatBuddy: That’s awesome! 😄 Keep being you!")
            elif "sad" in user_input:
                print("ChatBuddy: I'm here for you if you need a virtual hug. 🤗")
                offer_joke()
            elif "tired" in user_input:
                print("ChatBuddy: Don’t push yourself too hard. Rest is key. 😴")
            elif "hungry" in user_input:
                print("ChatBuddy: You deserve a snack! 🍪")
        else:
            # LEARNING MODE
            print("ChatBuddy: I don't know how to respond to that yet. 🤔")
            print("ChatBuddy: What should I say next time someone says that?")
            new_response = input(f"{user_name}: ")
            learned_memory[user_input] = new_response
        
            # Save to file
            with open(learned_memory_file, "w") as f:
                json.dump(learned_memory, f, indent=4)
        
            print("ChatBuddy: Got it! I'll remember that next time. 🧠")

📝 Try It For Yourself

Now that the json file & the chatbot.py file are updated, go ahead and start your bot up as usual!

Test the implementation by first going through the introduction phase (i.e. "What's your name?" & "How are you?")

Next, type something that the bot shouldn't recognize. Let's do something like this:

I like turtles

Your bot should then respond with something like:

ChatBuddy: I don't know how to respond to that yet. 🤔
ChatBuddy: What should I say next time someone says that?

Now, it's completely up to you as to what you want the response to be! For the sake of the exercise, let's say you go with something like:

That’s cool! Turtles are awesome! 🐢

After you tell the bot how it should respond, it should say:

ChatBuddy: Got it! I'll remember that next time. 🧠

Now go ahead and type in the same phrase as before. In this case:

I like turtles

And would you look at that! Your bot remembers!

If you go check your json file, you should now see something like this:

{
    "i like turtles": "That's cool! Turtles are awesome! 🐢"
}

Now, even if you end the conversation and restart the terminal, your bot will still remember the response you wanted!

🌟 If you ever want to change the response given or erase a certain memory (AKA perform bot brain surgery), all you have to do is head into your json file and adjust accordingly!
